41/// Parameters for the @ref BroydenGood class.
42struct BroydenGoodParams {
43 /// Length of the history to keep.
44 length_t memory = 10;
45 /// Reject update if @f$ s^\top Hy \le \text{min_div_fac} @f$.
46 real_t min_div_abs = 1e-32;
47 /// If set to true, the inverse Jacobian estimate should remain definite.
48 bool force_pos_def = false;
49 /// If set to true, the buffer is cleared after @p memory iterations. If
50 /// set to false, a circular buffer is used, replacing the oldest pair of
51 /// vectors, as a limited-memory type algorithm. However, since each vector
52 /// s̃ depends on the previous vectors, this algorithm is not theoretically
53 /// correct, because the s̃ vectors are not updated when the old history is
54 /// overwritten.
55 bool restarted = true;
56 /// Powell's trick, damping, prevents nonsingularity.
57 real_t powell_damping_factor = 0;
58 /// Minimum automatic step size. If @f$ \frac{s^\top y}{y^\top y} @f$ is
59 /// smaller than this setting, use this as the step size instead.
60 real_t min_stepsize = 1e-10;
61};

63/**
64 * Broyden's “Good” method for solving systems of nonlinear equations
65 * @f$ F(x) = 0 @f$.
66 *
67 * @f[ \begin{aligned}
68 * B_{k+1} &= B_k + \frac{\left(y_k - B_k s_k\right) s_k^\top}
69 * {s_k^\top s_k} \\
70 * H_{k+1} &= H_k + \frac{\left(s_k - H_k y_k\right)
71 * \left(s_k^\top H_k\right)}
72 * {s_k^\top H_k y_k} \\
73 * s_k &\triangleq x_{k+1} - x_k \\
74 * y_k &\triangleq F(x_{k+1}) - F(x_k) \\
75 * \end{aligned} @f]
76 * Where @f$ B_k @f$ approximates the Jacobian of @f$ F(x_k) @f$ and
77 * @f$ H_k \triangleq B_k^{-1} @f$.
78 *
79 * @todo Damping.
80 *
81 * @ingroup accelerators-grp
82 */
83class BroydenGood {

176template <class VecS, class VecY>
177bool BroydenGood::update_sy(const anymat<VecS> &sₖ, const anymat<VecY> &yₖ,
178 bool forced) {
179 // Restart if the buffer is full
180 if (full && params.restarted) {
181 full = false;
182 assert(idx == 0);
183 }
184
185 auto &&r = sto.work();
186 // Compute r = r₍ₘ₋₁₎ = Hₖ yₖ
187 r = yₖ; // r₍₋₁₎ = yₖ
188 foreach_fwd([&](index_t i) {
189 r += s̃(i) * r.dot(s(i)); // r₍ᵢ₎ = r₍ᵢ₋₁₎ + s̃₍ᵢ₎〈r₍ᵢ₋₁₎, s₍ᵢ₎〉
190 });
191 const real_t sᵀHy = sₖ.dot(r);
192 const real_t a_sᵀHy = params.force_pos_def ? sᵀHy : std::abs(sᵀHy);
193
194 // Check if update is accepted
195 if (!forced && a_sᵀHy < params.min_div_abs)
196 return false;
197
198 // Compute damping
199 real_t damp = 1 / sᵀHy;
200 if (real_t θ̅ = params.powell_damping_factor) {
201 real_t γ = sᵀHy / sₖ.squaredNorm();
202 real_t sgn_γ = γ >= 0 ? 1 : -1;
203 real_t γθ̅ = params.force_pos_def ? γ * θ̅ : sgn_γ * θ̅;
204 real_t a_γ = params.force_pos_def ? γ : std::abs(γ);
205 real_t θ = a_γ >= θ̅ ? 1 // no damping
206 : (1 - γθ̅) / (1 - γ);
207 damp = θ / (sₖ.squaredNorm() * (1 - θ + θ * γ));
208 }
209
210 // Store the new vectors
211 sto.s(idx) = sₖ;
212 sto.s̃(idx) = damp * (sₖ - r);
213 latest_γ = sₖ.dot(yₖ) / yₖ.squaredNorm();
214 if (std::abs(latest_γ) < params.min_stepsize)
215 latest_γ = std::copysign(params.min_stepsize, latest_γ);
216
217 // Increment the index in the circular buffer
218 idx = succ(idx);
219 full |= idx == 0;
220
221 return true;
222}

224inline bool BroydenGood::apply(rvec q, real_t γ) {
225 // Only apply if we have previous vectors s and y
226 if (idx == 0 && not full)
227 return false;
228
229 if (γ < 0)
230 γ = latest_γ;
231 if (γ != 1)
232 q *= γ;
233
234 // Compute q = q₍ₘ₋₁₎ = Hₖ q
235 // q₍₋₁₎ = q
236 foreach_fwd([&](index_t i) {
237 q += s̃(i) * q.dot(s(i)); // q₍ᵢ₎ = q₍ᵢ₋₁₎ + s̃₍ᵢ₎〈q₍ᵢ₋₁₎, s₍ᵢ₎〉
238 });
239
240 return true;
241}
